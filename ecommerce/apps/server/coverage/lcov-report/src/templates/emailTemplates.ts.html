const escapeHtml = (value: string): string =>
  value
    .replace(/&/g, '&amp;')
    .replace(/</g, '&lt;')
    .replace(/>/g, '&gt;')
    .replace(/"/g, '&quot;')
    .replace(/'/g, '&#39;');

const escapeUrl = (value: string): string => escapeHtml(value);

export const renderBaseTemplate = (title: string, body: string): string => `
  <div style="font-family: 'DM Sans', Arial, sans-serif; background: #0A1F44; padding: 32px; color: #ffffff;">
    <div style="max-width: 640px; margin: 0 auto; background: rgba(255,255,255,0.05); border: 1px solid rgba(255,255,255,0.1); border-radius: 20px; overflow: hidden;">
      <div style="padding: 24px 32px; background: linear-gradient(90deg, #D4AF37, #b8971f); color: #1A1A1A;">
        <h1 style="margin: 0; font-family: 'Playfair Display', Georgia, serif; font-size: 30px;">${escapeHtml(title)}</h1>
      </div>
      <div style="padding: 32px;">
        ${body}
      </div>
    </div>
  </div>
`;

const renderTextParagraphs = (value: string): string =>
  value
    .trim()
    .split(/\n\s*\n/)
    .filter(Boolean)
    .map((paragraph) => `<p>${escapeHtml(paragraph).replace(/\n/g, '<br />')}</p>`)
    .join('');

export const renderAdminBroadcastEmail = (payload: {
  previewText?: string;
  headline: string;
  body: string;
  ctaLabel?: string;
  ctaUrl?: string;
  audienceLabel?: string;
}): string =>
  renderBaseTemplate(
    payload.headline,
    `
      ${payload.audienceLabel ? `<p style="margin-top: 0; font-size: 12px; letter-spacing: 0.24em; text-transform: uppercase; color: #D4AF37;">${escapeHtml(payload.audienceLabel)}</p>` : ''}
      ${payload.previewText ? `<p style="font-size: 18px; line-height: 1.7; color: rgba(255,255,255,0.9);">${escapeHtml(payload.previewText)}</p>` : ''}
      <div style="font-size: 16px; line-height: 1.8; color: rgba(255,255,255,0.88);">
        ${renderTextParagraphs(payload.body)}
      </div>
      ${
        payload.ctaLabel && payload.ctaUrl
          ? `
            <p style="margin: 32px 0 0;">
              <a href="${escapeUrl(payload.ctaUrl)}" style="display: inline-block; padding: 12px 20px; border-radius: 999px; background: linear-gradient(90deg, #D4AF37, #b8971f); color: #1A1A1A; text-decoration: none; font-weight: 700;">
                ${escapeHtml(payload.ctaLabel)}
              </a>
            </p>
          `
          : ''
      }
      <p style="margin-top: 32px; font-size: 13px; color: rgba(255,255,255,0.66);">
        You are receiving this update because you are subscribed to NJ Store communications or have a verified NJ Store account.
      </p>
    `
  );
